import re import json from js import document, console, FileReader, alert, window from pyodide.ffi import create_proxy import asyncio class ATSAnalyzer: def __init__(self): self.keywords_tech = [ 'python', 'java', 'javascript', 'react', 'node.js', 'sql', 'html', 'css', 'git', 'aws', 'docker', 'kubernetes', 'machine learning', 'data analysis', 'project management', 'agile', 'scrum', 'leadership', 'communication' ] self.keywords_action = [ 'achieved', 'improved', 'increased', 'reduced', 'managed', 'led', 'developed', 'created', 'implemented', 'designed', 'analyzed', 'optimized', 'streamlined', 'coordinated', 'collaborated', 'delivered', 'executed', 'established' ] self.keywords_soft = [ 'teamwork', 'leadership', 'communication', 'problem-solving', 'analytical', 'creative', 'detail-oriented', 'self-motivated', 'adaptable', 'organized' ] def analyze_resume(self, text): text_lower = text.lower() # Calculate different scoring criteria criteria = {} # 1. Keyword Optimization (25 points) tech_matches = sum(1 for keyword in self.keywords_tech if keyword in text_lower) action_matches = sum(1 for keyword in self.keywords_action if keyword in text_lower) soft_matches = sum(1 for keyword in self.keywords_soft if keyword in text_lower) total_keywords = tech_matches + action_matches + soft_matches keyword_score = min(25, total_keywords * 2) criteria['keywords'] = { 'score': keyword_score, 'max_score': 25, 'details': f'Found {total_keywords} relevant keywords (Tech: {tech_matches}, Action: {action_matches}, Soft: {soft_matches})' } # 2. Formatting and Structure (20 points) format_score = 0 format_details = [] # Check for sections sections = ['experience', 'education', 'skills', 'summary', 'objective'] found_sections = sum(1 for section in sections if section in text_lower) format_score += min(10, found_sections * 2) format_details.append(f'Found {found_sections}/5 standard sections') # Check for bullet points bullet_count = text.count('‚Ä¢') + text.count('-') + text.count('*') if bullet_count > 5: format_score += 5 format_details.append('Good use of bullet points') # Check for consistent formatting if len(text.split('\n')) > 10: format_score += 5 format_details.append('Well-structured content') criteria['formatting'] = { 'score': format_score, 'max_score': 20, 'details': '; '.join(format_details) } # 3. Quantifiable Achievements (15 points) numbers = re.findall(r'\d+%|\$\d+|\d+\+|\d+ years?|\d+ months?', text) quant_score = min(15, len(numbers) * 2) criteria['quantifiable'] = { 'score': quant_score, 'max_score': 15, 'details': f'Found {len(numbers)} quantifiable achievements' } # 4. Contact Information (10 points) contact_score = 0 contact_details = [] # Email check if re.search(r'\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b', text): contact_score += 3 contact_details.append('Email present') # Phone check if re.search(r'\(?\d{3}\)?[-.\s]?\d{3}[-.\s]?\d{4}', text): contact_score += 3 contact_details.append('Phone present') # LinkedIn check if 'linkedin' in text_lower: contact_score += 2 contact_details.append('LinkedIn present') # Location check if re.search(r'\b[A-Z][a-z]+,\s*[A-Z]{2}\b', text): contact_score += 2 contact_details.append('Location present') criteria['contact'] = { 'score': contact_score, 'max_score': 10, 'details': '; '.join(contact_details) if contact_details else 'Missing contact information' } # 5. Length and Readability (10 points) word_count = len(text.split()) length_score = 0 if 200 <= word_count <= 800: length_score = 10 length_details = f'Optimal length ({word_count} words)' elif word_count < 200: length_score = 5 length_details = f'Too short ({word_count} words)' else: length_score = 7 length_details = f'Acceptable length ({word_count} words)' criteria['length'] = { 'score': length_score, 'max_score': 10, 'details': length_details } # 6. ATS-Friendly Elements (10 points) ats_score = 0 ats_details = [] # No special characters in excess special_chars = len(re.findall(r'[^\w\s\-\.\,\:\;\!\?\(\)]', text)) if special_chars < 10: ats_score += 3 ats_details.append('Minimal special characters') # Standard fonts (assumed from simple text) ats_score += 2 ats_details.append('Text-based format') # No tables or complex formatting (assumed) ats_score += 3 ats_details.append('Simple structure') # Standard section headers if any(header in text_lower for header in ['experience', 'education', 'skills']): ats_score += 2 ats_details.append('Standard headers') criteria['ats_friendly'] = { 'score': ats_score, 'max_score': 10, 'details': '; '.join(ats_details) } # 7. Skills Section Quality (10 points) skills_score = 0 skills_details = [] if 'skills' in text_lower: skills_score += 5 skills_details.append('Skills section present') # Check for variety of skill types skill_types = 0 if any(tech in text_lower for tech in self.keywords_tech[:10]): skill_types += 1 if any(soft in text_lower for soft in self.keywords_soft[:5]): skill_types += 1 skills_score += skill_types * 2 skills_details.append(f'Multiple skill categories') criteria['skills'] = { 'score': skills_score, 'max_score': 10, 'details': '; '.join(skills_details) if skills_details else 'No skills section found' } # Calculate total score total_score = sum(criteria[key]['score'] for key in criteria) max_total = sum(criteria[key]['max_score'] for key in criteria) return { 'total_score': total_score, 'max_score': max_total, 'percentage': round((total_score / max_total) * 100), 'criteria': criteria, 'suggestions': self.generate_suggestions(criteria) } def generate_suggestions(self, criteria): suggestions = [] if criteria['keywords']['score'] < 15: suggestions.append("Add more relevant keywords for your industry and role") if criteria['formatting']['score'] < 15: suggestions.append("Improve formatting with clear sections and bullet points") if criteria['quantifiable']['score'] < 10: suggestions.append("Include more quantifiable achievements (percentages, numbers, metrics)") if criteria['contact']['score'] < 8: suggestions.append("Ensure all contact information is present and correctly formatted") if criteria['length']['score'] < 8: suggestions.append("Adjust resume length to 200-800 words for optimal ATS scanning") if criteria['skills']['score'] < 7: suggestions.append("Add a dedicated skills section with relevant technical and soft skills") if criteria['ats_friendly']['score'] < 8: suggestions.append("Use standard formatting and avoid complex layouts or special characters") return suggestions # Initialize analyzer analyzer = ATSAnalyzer() def analyze_resume(event): # Show loading document.getElementById('loading').style.display = 'block' document.getElementById('results').style.display = 'none' # Get resume text resume_text = document.getElementById('resumeText').value if not resume_text.strip(): document.getElementById('loading').style.display = 'none' alert('Please paste your resume text or upload a file.') return # Analyze resume try: results = analyzer.analyze_resume(resume_text) display_results(results) except Exception as e: console.log(f"Error analyzing resume: {str(e)}") document.getElementById('loading').style.display = 'none' alert('Error analyzing resume. Please try again.') def display_results(results): # Hide loading document.getElementById('loading').style.display = 'none' # Show results document.getElementById('results').style.display = 'block' # Display score score = results['percentage'] document.getElementById('scoreValue').textContent = f"{score}%" # Set score color and description score_circle = document.getElementById('scoreCircle') score_desc = document.getElementById('scoreDescription') if score >= 80: score_circle.className = 'score-circle score-excellent' score_desc.textContent = 'Excellent ATS Compatibility!' elif score >= 60: score_circle.className = 'score-circle score-good' score_desc.textContent = 'Good ATS Compatibility' else: score_circle.className = 'score-circle score-fair' score_desc.textContent = 'Needs Improvement' # Display criteria breakdown criteria_grid = document.getElementById('criteriaGrid') criteria_grid.innerHTML = '' criteria_names = { 'keywords': 'Keyword Optimization', 'formatting': 'Formatting & Structure', 'quantifiable': 'Quantifiable Achievements', 'contact': 'Contact Information', 'length': 'Length & Readability', 'ats_friendly': 'ATS-Friendly Elements', 'skills': 'Skills Section Quality' } for key, data in results['criteria'].items(): criteria_item = document.createElement('div') criteria_item.className = 'criteria-item' score_percent = round((data['score'] / data['max_score']) * 100) score_class = 'score-green' if score_percent >= 80 else 'score-orange' if score_percent >= 60 else 'score-red' criteria_item.innerHTML = f"""
{criteria_names[key]}
{data['score']}/{data['max_score']} ({score_percent}%)
{data['details']}
""" criteria_grid.appendChild(criteria_item) # Display suggestions suggestions_list = document.getElementById('suggestionsList') suggestions_list.innerHTML = '' for suggestion in results['suggestions']: li = document.createElement('li') li.textContent = suggestion suggestions_list.appendChild(li) # File handling with PDF support def handle_file_select(event): files = event.target.files if files and files.length > 0: file = files.item(0) file_name = file.name.lower() document.getElementById('fileName').textContent = f"Selected: {file.name}" # Process different file types if file_name.endswith('.pdf'): process_pdf_file(file) elif file_name.endswith('.txt'): process_text_file(file) else: alert('Please upload a PDF or TXT file, or paste your text directly.') def process_pdf_file(file): """Process PDF file using JavaScript helper""" document.getElementById('fileName').textContent = f"Processing PDF: {file.name}..." def on_pdf_success(text_content): if text_content and text_content.strip(): document.getElementById('resumeText').value = text_content document.getElementById('fileName').textContent = f"‚úÖ PDF loaded: {file.name}" console.log("PDF processed successfully") else: document.getElementById('fileName').textContent = f"‚ùå Could not extract text from: {file.name}" alert('Could not extract text from PDF. Please try pasting the text directly.') def on_pdf_error(error): console.log(f"PDF processing error: {error}") document.getElementById('fileName').textContent = f"‚ùå Error processing: {file.name}" alert('Error processing PDF file. Please try pasting the text directly.') # Use JavaScript helper function try: promise = window.extractPDFText(file) promise.then(create_proxy(on_pdf_success)).catch(create_proxy(on_pdf_error)) except Exception as e: console.log(f"Error calling PDF extraction: {str(e)}") on_pdf_error(str(e)) def process_text_file(file): """Process text file""" reader = FileReader.new() def on_load(event): text_content = event.target.result document.getElementById('resumeText').value = text_content document.getElementById('fileName').textContent = f"‚úÖ Text file loaded: {file.name}" def on_error(event): console.log("Error reading text file") document.getElementById('fileName').textContent = f"‚ùå Error reading: {file.name}" alert('Error reading text file. Please try pasting the text directly.') reader.onload = create_proxy(on_load) reader.onerror = create_proxy(on_error) reader.readAsText(file) # Event listeners document.getElementById('analyzeBtn').addEventListener('click', create_proxy(analyze_resume)) document.getElementById('fileInput').addEventListener('change', create_proxy(handle_file_select))
